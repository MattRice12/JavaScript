var object1 = {length: 5, unit: "meter"};

  *Modify Properties* 
    object1.length += 2
      #=> length: 7

  *Add Properties*
    object1.color = "red"
      #=> color: red

  *Delete Properties*
    delete object1.unit;


# Bracket Notation
  object1["num of vertices"] = 5;
